load_tester/
│
├── core/
│   ├── main.py              # entry point (starts master)
│   ├── master.py            # orchestration layer
│   ├── worker.py            # single worker process (event loop owner)
│   ├── event_loop.py        # epoll wrapper
│   ├── fd_registry.py       # fd → connection mapping
│
├── client/
│   ├── client_connection.py # per-client state machine
│   ├── client_pool.py       # creates and tracks M clients
│   ├── client_generator.py  # request generator
│
├── network/
│   ├── socket_utils.py      # non-blocking socket helpers
│   ├── reader.py            # recv + parsing logic
│   ├── writer.py            # send logic
│
├── protocol/
│   ├── protocol.py          # constants (header format, magic bytes)
│   ├── serializer.py        # encode request
│   ├── parser.py            # decode response
│
├── state/
│   ├── client_state.py      # CONNECTING/WRITING/READING/DONE
│
├── metrics/
│   ├── stats.py             # latency, success, failure tracking
│
├── config/
│   ├── config.py            # host, port, M, buffer sizes
│
├── utils/
│   ├── logger.py            # logging wrapper
│   ├── time_utils.py        # timestamps
│
└── README.md
